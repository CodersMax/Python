import flask
from flask import request, jsonify

app = flask.Flask(__name__)

@app.route('/', methods=['GET'])
def home():
    return '''<html><body bgcolor="blue"><h1>PROVIDE SOME DATA</h1><br><br><p>use the argument    http://IPaddress:5000/api?data=SOME_DATA</p></body></html>'''
    
@app.route('/api', methods=['GET'])
def api_data():
    if 'data' not in request.args:
        return '''<html><body bgcolor="blue"><h1>PROVIDE SOME DATA</h1><br><br><p>use the argument    http://IPaddress:5000/api?data=SOME_DATA</p></body></html>'''
    if 'data' in request.args:
        data = request.args['data']
        print("")
        print("======")
        print("")
        print("SENDED DATA  :", data)
        print("")
        print("======")
        print("")
    else:
        return """<html><body bgcolor="red"><b><h1>Data Sending Process Failed</h1></b></body></html>"""
        
    return """<html><body bgcolor="green"><b><h1>Data Sending Process Success</h1></b></body></html>"""
    
app.run()
